import streamlit as st
import geopandas as gpd
import pydeck as pdk
from pathlib import Path
from data_utils import load_project_data

st.title("Congestion Segments Map")

GDB_PATH = Path("data/Cohort3_CongestionSegments.gdb")
LAYER_NAME = "Cohort3onXD2501"


@st.cache_data
def load_gis_data():
    if not GDB_PATH.exists():
        raise FileNotFoundError(f"GDB not found: {GDB_PATH}")

    gdf = gpd.read_file(GDB_PATH, layer=LAYER_NAME)

    # Reproject to WGS84 for web mapping
    if gdf.crs is not None and str(gdf.crs) != "EPSG:4326":
        gdf = gdf.to_crs(epsg=4326)

    return gdf


@st.cache_data
def load_joined_data():
    gdf = load_gis_data()
    project_df = load_project_data().copy()

    # Normalize join keys
    if "SMART_SCALE_ID" in gdf.columns:
        gdf["SMART_SCALE_ID"] = gdf["SMART_SCALE_ID"].astype(str).str.strip()

    if "AppID" in project_df.columns:
        project_df["AppID"] = project_df["AppID"].astype(str).str.strip()

    # Keep only useful project columns for hover / filtering
    project_cols = [
        "AppID",
        "Title",
        "District",
        "FacilityType",
        "DominantImprovementType",
        "AllImprovementTypes",
        "ImprovementTypeCount",
        "AvgDelayBefore",
        "AvgDelayAfter",
        "AvgDelayChange",
        "AvgSpeedBefore",
        "AvgSpeedAfter",
        "AvgSpeedChange",
        "PTIChange",
        "DataQualityFlag",
        "ConfounderFlag",
    ]
    project_cols = [c for c in project_cols if c in project_df.columns]

    project_info = project_df[project_cols].drop_duplicates(subset=["AppID"])

    # Join GIS layer with project info
    if "SMART_SCALE_ID" in gdf.columns and "AppID" in project_info.columns:
        gdf = gdf.merge(
            project_info,
            left_on="SMART_SCALE_ID",
            right_on="AppID",
            how="left"
        )

    return gdf


try:
    gdf = load_joined_data()
except Exception as e:
    st.error(f"Failed to load GIS layer: {e}")
    st.stop()

st.write(f"Layer: **{LAYER_NAME}**")
st.write(f"Features: **{len(gdf)}**")
st.write(f"CRS: **{gdf.crs}**")

st.sidebar.header("GIS Controls")

filtered_gdf = gdf.copy()

# -------------------------
# Project ID filter
# -------------------------
if "SMART_SCALE_ID" in filtered_gdf.columns:
    project_ids = (
        filtered_gdf["SMART_SCALE_ID"]
        .dropna()
        .astype(str)
        .unique()
        .tolist()
    )
    project_ids = sorted(project_ids)
else:
    project_ids = []

selected_project_ids = st.sidebar.multiselect("Project ID", project_ids)

if selected_project_ids:
    filtered_gdf = filtered_gdf[
        filtered_gdf["SMART_SCALE_ID"].astype(str).isin(selected_project_ids)
    ]

# -------------------------
# Optional attribute filter
# -------------------------
non_geom_cols = [c for c in filtered_gdf.columns if c != "geometry"]

selected_filter_col = st.sidebar.selectbox(
    "Filter by attribute (optional)",
    options=["None"] + non_geom_cols
)

if selected_filter_col != "None":
    unique_vals = (
        filtered_gdf[selected_filter_col]
        .dropna()
        .astype(str)
        .unique()
        .tolist()
    )
    unique_vals = sorted(unique_vals)

    if len(unique_vals) <= 100:
        selected_vals = st.sidebar.multiselect(
            f"Select {selected_filter_col}",
            unique_vals
        )
        if selected_vals:
            filtered_gdf = filtered_gdf[
                filtered_gdf[selected_filter_col].astype(str).isin(selected_vals)
            ]
    else:
        st.sidebar.info("Too many unique values to list.")

line_width = st.sidebar.slider("Line width", min_value=1, max_value=12, value=4)
show_table = st.sidebar.checkbox("Show attribute table", value=False)
show_no_basemap = st.sidebar.checkbox("Use no basemap (debug)", value=False)

st.write(f"Filtered features: **{len(filtered_gdf)}**")

if len(filtered_gdf) == 0:
    st.warning("No features match the current filter.")
    st.stop()

# -------------------------
# Debug info
# -------------------------
st.subheader("Debug Information")

bounds = filtered_gdf.total_bounds.tolist()
st.write(f"Bounds: {bounds}")

minx, miny, maxx, maxy = filtered_gdf.total_bounds
center_lon = (minx + maxx) / 2
center_lat = (miny + maxy) / 2

st.write("Map center:", {"latitude": center_lat, "longitude": center_lon})

# -------------------------
# Tooltip fields
# -------------------------
tooltip_lines = []

preferred_tooltip_fields = [
    "SMART_SCALE_ID",
    "Title",
    "District",
    "FacilityType",
    "DominantImprovementType",
    "AllImprovementTypes",
    "ImprovementTypeCount",
    "ANALYSIS_TYPE",
    "FACILITY_NAME",
    "DIRECTION",
    "XD_SEGMENT_ID",
    "AvgDelayChange",
    "AvgSpeedChange",
    "PTIChange",
    "DataQualityFlag",
    "ConfounderFlag",
]

for col in preferred_tooltip_fields:
    if col in filtered_gdf.columns:
        tooltip_lines.append(f"{col}: {{{col}}}")

tooltip_text = "\n".join(tooltip_lines) if tooltip_lines else "Segment"

geojson_data = filtered_gdf.__geo_interface__

layer = pdk.Layer(
    "GeoJsonLayer",
    data=geojson_data,
    pickable=True,
    stroked=True,
    filled=False,
    get_line_color=[255, 0, 0],
    get_line_width=line_width,
    line_width_min_pixels=2,
)

view_state = pdk.ViewState(
    latitude=center_lat,
    longitude=center_lon,
    zoom=6,
    pitch=0
)

deck = pdk.Deck(
    map_style=None if show_no_basemap else "light",
    layers=[layer],
    initial_view_state=view_state,
    tooltip={"text": tooltip_text},
)

st.pydeck_chart(deck, use_container_width=True)

# -------------------------
# Attribute table
# -------------------------
st.subheader("Attribute Summary")
st.write("Columns:")
st.write(non_geom_cols)

if show_table:
    st.subheader("Attribute Table Preview")
    preview_cols = [c for c in non_geom_cols if c != "geometry"][:15]
    st.dataframe(filtered_gdf[preview_cols].head(50), use_container_width=True)
